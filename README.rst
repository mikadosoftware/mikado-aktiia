Plugin 'puremourning/vimspector'
and 
" from vimspector site - https://puremourning.github.io/vimspector-web/demo-setup.html
let g:vimspector_enable_mappings = 'HUMAN'

And :PluginInstall

Then find a file and click F5

then choose basic run
